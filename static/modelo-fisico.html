Modelo Físico do Banco de Dados — CondConnect (MySQL)
usuarios
id INT PK AUTO_INCREMENT
nome VARCHAR(100)
email VARCHAR(100) UNIQUE
senha VARCHAR(255)
foto_url VARCHAR(255)
papel ENUM('membro','admin')
pix_key VARCHAR(100)
rating DECIMAL(3,2)
data_criacao DATETIME
produtos
id INT PK AUTO_INCREMENT
titulo VARCHAR(100)
descricao TEXT
categoria VARCHAR(50)
preco DECIMAL(10,2)
condicao ENUM('Novo','Seminovo',
'Usado')
status ENUM('ativo','vendido','inativo')
data_publicacao DATETIME
id_vendedor INT FK → usuarios.id
favoritos
id INT PK AUTO_INCREMENT
usuario_id INT FK → usuarios.id
produto_id INT FK → produtos.id
data_adicao DATETIME
notificacoes
id INT PK AUTO_INCREMENT
id_usuario INT FK → usuarios.id
titulo VARCHAR(100)
mensagem VARCHAR(255)
lida BOOLEAN
data_envio DATETIME
imagens_produto
id INT PK AUTO_INCREMENT
produto_id INT FK → produtos.id
url_imagem VARCHAR(255)
principal BOOLEAN
pedidos
id INT PK AUTO_INCREMENT
produto_id INT FK → produtos.id
comprador_id INT FK → usuarios.id
vendedor_id INT FK → usuarios.id
status ENUM('pendente','enviado',
'entregue','cancelado')
data_pedido DATETIME
valor DECIMAL(10,2)
propostas
id INT PK AUTO_INCREMENT
produto_id INT FK → produtos.id
comprador_id INT FK → usuarios.id
vendedor_id INT FK → usuarios.id
valor DECIMAL(10,2)
quantidade INT
status ENUM('pendente','aceita',
'recusada')
data_proposta DATETIME
itens_carrinho
id INT PK AUTO_INCREMENT
usuario_id INT FK → usuarios.id
produto_id INT FK → produtos.id
preco_negociado DECIMAL(10,2)
avaliacoes
id INT PK AUTO_INCREMENT
produto_id INT FK → produtos.id
usuario_id INT FK → usuarios.id
nota INT (1–5)
comentario TEXT
data_avaliacao DATETIME
PK — Chave Primária
FK — Chave Estrangeira
(1,N) — Um para Muitos
